#include <sys/socket.h>
#include <netdb.h>
#include <arpa/inet.h>

//this file tell you how to create a static router
//this function has been abandoned, but must do the compatible
void * Exp_CreateRout(const char *ifname)
{
    void *ret = NULL;
    tagHostAddrV4 v4;
    tagRouterPara routpara;
    memset(&routpara,0,sizeof(para));
    memset(&v4,0,sizeof(v4));
    v4.host = inet_addr("192.168.0.100");//our host ip
    v4.net = inet_addr("192.168.0.0");//our net
    v4.mask = inet_addr("255.255.255.0");//our net mask
    v4.broad = inet_addr("192.168.0.255"); //our net broad

    routpara.ver = EN_IPV_4;
    routpara.host = &v4.ip;
    routpara.mask = &v4.mask;
    routpara.broad = &v4.broad;
    routpara.net = &v4.net;
    routpara.prior = CN_ROUT_PRIOR_UNI;
    routpara.ifname = ifname;
    ret = RouterCreate(&para);
    return ret;
}
